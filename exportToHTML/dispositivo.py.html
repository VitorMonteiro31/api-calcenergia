from fastapi import APIRouter
from models.dependencia
from models.dispositivo
from models.unidade_consumidora
from schemas.dispositivos
    DispositivoCreate,
    DispositivoRead,
    DispositivoReadList,
    DispositivoUpdate
)

router = APIRouter

@router.post
def criar_dispositivo(dispositivo: DispositivoCreate):
    dispositivo.nome = dispositivo.nome
    dispositivo.consumo = dispositivo.consumo
    dispositivo.uso_diario = dispositivo.uso_diario
    dispositivo.tipo_id = dispositivo.tipo_id
    dispositivo.save()
    return dispositivo

@router.get
path='/unidades-consumidoras/{unidade_consumidora_id}'
response_model=DispositivoReadList
)
def listar_dispositivos(unidade_consumidora_id: int):
    dispositivos = DispositivoDB.find(unidade_consumidora_id)
    return {'dispositivos': dispositivos}

@router.get
path='/dependencias/{dependencia_id}'
)
def listar_dispositivos_por_dependencia(dependencia_id: int):
    dispositivos = DispositivoDB.find_by_dependencia(dependencia_id)
    return {'dispositivos': dispositivos}

@router.get
path='/dispositivos/{dispositivo_id}'
)
def listar_dispositivo(dispositivo_id: int):
    dispositivo = DispositivoDB.find(dispositivo_id)
    return dispositivo

@router.patch
def atualizar_dispositivo(id_dispositivo: int, dados: DispositivoUpdate):
    dispositivo = DispositivoDB.update(id_dispositivo, dados)
    return dispositivo

@router.delete
def eliminar_dispositivo(id_dispositivo: int):
    dispositivo = DispositivoDB.find(id_dispositivo)
    dispositivo.delete_instance()
    return dispositivo
